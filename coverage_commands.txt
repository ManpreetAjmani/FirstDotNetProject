docker run --name jenkins-blueocean --rm --detach --network jenkins --env DOCKER_HOST=tcp://docker:2376 --env DOCKER_CERT_PATH=/certs/client --env DOCKER_TLS_VERIFY=1 --volume jenkins-data:/var/jenkins_home -d --volume jenkins-docker-certs:/certs/client:ro --publish 9090:8080 --publish 50000:50000 jenkinsci/blueocean

docker run -d --name sonarqube -p 9000:9000 -p 9092:9092 --network jenkins -v sonarqube-data:/opt/sonarqube/data -v sonarqube-extensions:/opt/sonarqube/extensions sonarqube
